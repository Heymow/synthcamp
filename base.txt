        const StatusTimer = ({ baseTime, isCountdown = false, small = false }) => {
            const [time, setTime] = useState("");

                const update = () => {
                    const now = Date.now();
                    const diff = isCountdown ? baseTime - now : now - baseTime;
                    if (isCountdown && diff <= 0) { setTime("LIVE"); return; }
                    const seconds = Math.floor((Math.abs(diff) / 1000) % 60);
                    const minutes = Math.floor((Math.abs(diff) / (1000 * 60)) % 60);
                    const hours = Math.floor((Math.abs(diff) / (1000 * 60 * 60)) % 24);
                    const parts = [
                        hours > 0 ? hours.toString().padStart(2, '0') : null,
                        minutes.toString().padStart(2, '0'),
                        seconds.toString().padStart(2, '0')
                    ].filter(Boolean);
                    setTime(parts.join(":"));
                };
                update();
                const interval = setInterval(update, 1000);
                return () => clearInterval(interval);
            }, [baseTime, isCountdown]);

            return (
                <div className={`absolute z-20 pointer-events-none ${small ? 'top-2 right-4' : 'top-5 right-5'}`}>
                    <span className={`${small ? 'text-[7px]' : 'text-[8px]'} font-black uppercase tracking-[0.15em] bg-black/60 backdrop-blur-md px-2.5 py-1.5 rounded-full border border-white/10 text-white/50 inline-flex items-center`}>
